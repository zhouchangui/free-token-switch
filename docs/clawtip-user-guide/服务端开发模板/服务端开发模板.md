package com.example.demo.controller;

import cn.hutool.core.codec.Base64;
import com.example.demo.crypto.SM4EncryptionUtils;
import com.example.demo.domain.OrderEntity;
import com.example.demo.domain.FulfillmentRecordEntity;
import com.example.demo.repository.OrderRepository;
import com.example.demo.repository.FulfillmentRecordRepository;
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.web.bind.annotation.*;

import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.*;

/**
 * 商家侧示例 Controller（只演示 createOrder 与 getServiceResult）。
 *
 * 重要说明：
 * 1) 这个示例对接的是“支付后获取服务结果”的典型流程，不局限于天气服务。
 *    你可以把 serviceResult 换成任何数字商品/数字服务内容，例如：
 *    - 报告摘要
 *    - 数据下载链接
 *    - AI 生成结果
 *    - 第三方接口调用额度发放结果
 *
 * 2) createOrder 接口返回订单参数给前端/调用方后，商家必须在自己数据库中落库订单关键信息，
 *    以便后续履约、对账、售后、风控、审计。示例里给出了建议落库字段。
 *
 * 3) getServiceResult。 接口中“解密支付凭证 -> 判断 SUCCESS -> 查询数据库履约状态 -> 决定是否履约”
 *    是核心。此示例强调“一次服务一次履约”，通过数据库幂等控制避免重复发放。
 */
@RestController
@RequestMapping("/api/demo")
@RequiredArgsConstructor
@Slf4j
public class MerchantDemoController {

    private final OrderRepository orderRepository;
    private final FulfillmentRecordRepository fulfillmentRecordRepository;
    private final ObjectMapper objectMapper = new ObjectMapper();


    /**
     * 加密密钥
     *  sm4key 为您在邮件里收到的 安全密钥 sm4key
     */
    private String sm4Key;

    /**
     * 收款方账号
     * 为您在邮件里收到的  收款凭证ID payTo
     */
    private static final String PAY_TO;

    /**
     * 创建订单接口。
     *
     * 服务端关键动作说明：
     * - 该接口返回 orderNo / amount / payTo / encryptedData 给调用方后，
     *   服务端“必须”自行将订单数据落库（示例里已经示意写入 orderRepository）。
     * - 后续 getServiceResult 是否给服务，依赖这条订单记录以及履约记录。
     */
    @PostMapping("/createOrder")
    public Map<String, Object> createOrder(@RequestBody Map<String, String> request) {
        Map<String, Object> resp = new LinkedHashMap<>();

        try {
            String question = request.get("question");
            String clientIp = request.get("clientIp");

            if (question == null || question.trim().isEmpty()) {
                resp.put("responseCode", "400");
                resp.put("responseMessage", "Missing required parameter: question");
                return resp;
            }

            // 1) 生成订单号（这里我们的逻辑为：时间戳 + 6位随机数）（实际请以您的业务逻辑为准）
            String orderNo = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyyMMddHHmmss"))
                    + (new Random().nextInt(900000) + 100000);

            // 2) 订单金额（这里我们的逻辑为：1分）（实际请以您的业务逻辑为准）
            BigDecimal amount = new BigDecimal("1");

            // 3) 构建给调用方的加密报文（订单号/金额/收款方）
            Map<String, Object> plainDict = new HashMap<>();
            plainDict.put("orderNo", orderNo);
            plainDict.put("amount", amount);
            plainDict.put("payTo", PAY_TO);

            String plainText = objectMapper.writeValueAsString(plainDict);
            // 这里的加密工具，您可以参考使用：服务端SM4加解密代码示例.md 中的代码
            String encryptedData = SM4EncryptionUtils.encrypt(plainText, Base64.decode(sm4Key));

            if (encryptedData == null || encryptedData.isEmpty()) {
                resp.put("responseCode", "500");
                resp.put("responseMessage", "SM4 encrypt failed");
                return resp;
            }

            // 4) 商家侧订单落库（非常关键）
            //    说明：很多对接方只把 orderNo 返回给前端而不落库，后续就无法做履约幂等和审计。
            //    建议最少保存：
            //    - orderNo（唯一）
            //    - 业务请求参数（如 question 或对应业务输入）
            //    - amount、payTo
            //    - orderStatus（INIT / PAID / CLOSED 等）
            //    - fulfillStatus（UNFULFILLED / FULFILLED）
            //    - createdAt、updatedAt
            //    - clientIp（可选，便于风控）
            OrderEntity order = new OrderEntity();
            order.setOrderNo(orderNo);
            order.setQuestion(question);
            order.setAmount(amount);
            order.setPayTo(PAY_TO);
            order.setOrderStatus("INIT");
            order.setFulfillStatus("UNFULFILLED");
            order.setClientIp(clientIp);
            order.setCreatedAt(LocalDateTime.now());
            order.setUpdatedAt(LocalDateTime.now());
            orderRepository.save(order);

            // 5) 返回给调用方用于拉起支付或生成支付凭证
            resp.put("responseCode", "200");
            resp.put("responseMessage", "Success");
            resp.put("orderNo", orderNo);
            resp.put("amount", amount);
            resp.put("payTo", PAY_TO);
            resp.put("encryptedData", encryptedData);
            return resp;

        } catch (Exception e) {
            log.error("createOrder error", e);
            resp.put("responseCode", "500");
            resp.put("responseMessage", "createOrder failed: " + e.getMessage());
            return resp;
        }
    }

    /**
     * 获取履约结果接口（示例名沿用 getServiceResult）。
     *
     * 业务核心：
     * A. 解密 credential，解析 payStatus
     * B. 若支付未成功，不履约
     * C. 若支付成功，查数据库订单并判断是否已履约
     * D. 未履约 -> 提供一次服务 + 落库履约记录 + 更新订单履约标记
     * E. 已履约 -> 返回“已履约，不重复提供”
     *
     * 特别说明：
     * - 本示例用天气结果作为“服务内容”演示，但您可实际可替换为任意服务结果。
     * - 一次性服务场景一定要做幂等控制，否则会出现重复发放权益。
     */
    @PostMapping("/getServiceResult")
    public Map<String, Object> getServiceResult(@RequestBody Map<String, String> request) {
        Map<String, Object> resp = new LinkedHashMap<>();

        try {
            String orderNo = request.get("orderNo");
            String credential = request.get("credential");
            String question = request.get("question");

            if (orderNo == null || orderNo.trim().isEmpty()
                    || credential == null || credential.trim().isEmpty()
                    || question == null || question.trim().isEmpty()) {
                resp.put("responseCode", "400");
                resp.put("responseMessage", "Missing required parameters: orderNo, credential, question");
                resp.put("payStatus", "ERROR");
                resp.put("errorInfo", "缺少必要参数");
                return resp;
            }

            // 1) 查询订单是否存在（数据库为准）
            OrderEntity order = orderRepository.findByOrderNo(orderNo).orElse(null);
            if (order == null) {
                resp.put("responseCode", "404");
                resp.put("responseMessage", "订单不存在");
                resp.put("payStatus", "ERROR");
                resp.put("errorInfo", "orderNo not found");
                return resp;
            }

            // 2) 解密支付凭据并解析 payStatus
            // 这里的解密工具，您可以参考使用：服务端SM4加解密代码示例.md 中的代码
            String decrypted = SM4EncryptionUtils.decrypt(credential, Base64.decode(sm4Key));
            if (decrypted == null || decrypted.isEmpty()) {
                resp.put("responseCode", "200");
                resp.put("responseMessage", "Success");
                resp.put("payStatus", "ERROR");
                resp.put("errorInfo", "解密失败，请检查凭据");
                return resp;
            }

            String payStatus = "PENDING";
            try {
                JsonNode root = objectMapper.readTree(decrypted);
                if (root.has("payStatus")) {
                    payStatus = root.get("payStatus").asText("PENDING");
                }
            } catch (Exception parseEx) {
                log.warn("credential parse failed, orderNo={}", orderNo, parseEx);
            }

            // 3) 支付未成功：不履约
            if (!"SUCCESS".equalsIgnoreCase(payStatus)) {
                resp.put("responseCode", "200");
                resp.put("responseMessage", "Success");
                resp.put("payStatus", payStatus);
                resp.put("errorInfo", "支付未成功，暂不履约");
                return resp;
            }

            // 4) 支付成功后更新订单支付态（示例）
            // 您可根据finishedTime实际情况，决定是否认为该凭证为过期凭证，按照您的逻辑拒绝提供服务
            String finishedTime =  root.get("finishTime").asText();
            order.setFinishedTime(finishedTime);
            order.setOrderStatus("PAID");
            order.setUpdatedAt(LocalDateTime.now());
            orderRepository.save(order);

            // 5) 一次性服务履约判断（数据库幂等）
            //    你提到“原来用缓存实现，但应改数据库”，这里就是数据库实现的核心：
            //    - 先查履约标记
            //    - 未履约才发放
            //    - 发放后立即写履约记录并更新标记
            if ("FULFILLED".equalsIgnoreCase(order.getFulfillStatus())) {
                // 已履约：防重复发放
                FulfillmentRecordEntity record = fulfillmentRecordRepository
                        .findTopByOrderNoOrderByCreatedAtDesc(orderNo)
                        .orElse(null);

                resp.put("responseCode", "200");
                resp.put("responseMessage", "Success");
                resp.put("payStatus", "SUCCESS");
                resp.put("alreadyFulfilled", true);
                resp.put("errorInfo", "该订单已履约，不重复提供服务");
                if (record != null) {
                    // 返回历史履约内容，便于前端展示“你已领取过”
                    resp.put("answer", record.getServiceResult());
                    resp.put("fulfilledAt", record.getCreatedAt().toString());
                }
                return resp;
            }

            // 6) 未履约：执行一次服务发放
            //    这里演示天气结果。实际项目中请替换为你们真实业务服务产出。
            //    再强调一遍：这个结构并不绑定天气业务，任何“可交付结果”都可以放这里。
            String weatherResult = "{\n" +
                    "  \"status\":\"1\",\n" +
                    "  \"info\":\"OK\",\n" +
                    "  \"serviceType\":\"weather-demo\",\n" +
                    "  \"tips\":\"可替换为任意数字服务返回\",\n" +
                    "  \"data\":{\n" +
                    "    \"city\":\"北京市\",\n" +
                    "    \"weather\":\"晴\",\n" +
                    "    \"temperature\":\"18\"\n" +
                    "  }\n" +
                    "}";

            // 7) 记录履约明细（用于审计/售后/对账）
            FulfillmentRecordEntity record = new FulfillmentRecordEntity();
            record.setOrderNo(orderNo);
            record.setServiceResult(weatherResult);
            record.setServiceType("ONE_TIME_SERVICE");
            record.setCreatedAt(LocalDateTime.now());
            fulfillmentRecordRepository.save(record);

            // 8) 更新订单履约标记，确保后续请求不再重复发放
            order.setFulfillStatus("FULFILLED");
            order.setUpdatedAt(LocalDateTime.now());
            orderRepository.save(order);

            // 9) 返回本次履约结果
            resp.put("responseCode", "200");
            resp.put("responseMessage", "Success");
            resp.put("payStatus", "SUCCESS");
            resp.put("alreadyFulfilled", false);
            resp.put("answer", weatherResult);
            return resp;

        } catch (Exception e) {
            log.error("getWeatherResult error", e);
            resp.put("responseCode", "500");
            resp.put("responseMessage", "getWeatherResult failed: " + e.getMessage());
            resp.put("payStatus", "ERROR");
            return resp;
        }
    }
}
